# for emacs users

    * mkdir ~/.emacs.d
    * cp perltidy.el ~/.emacs.d
    * add to ~/.emacs
(require 'perltidy)
(global-set-key "\C-cB" 'perltidy-buffer)
(global-set-key "\C-cS" 'perltidy-subroutine)
(global-set-key "\C-cR" 'perltidy-region)

    * verify that you have ~/.perltidyrc
    * restart emacs
 
# There are 3 keybindings:
    * Ctrl-c B    -> complete file
    * Ctrl-c S    -> sub where cursor is placed
    * Ctrl-c R    -> selected region
